### クーリング・オフ（注意: 「読解という名の無賃労働」と「システムへの信頼の欠如」）

#### クーリング・オフとは

クーリング・オフは、消費者が一定期間内であれば、理由なく契約を解除できる制度です。

#### クーリング・オフが適用される取引

- **訪問販売**: 事業者の営業所等以外の場所で行われる販売
- **通信販売**: 郵便、電話、インターネット等による販売
- **電話勧誘販売**: 電話による勧誘による販売

#### クーリング・オフの期間

- **訪問販売**: 契約日から8日間
- **通信販売**: 商品受領日から8日間（サービス提供開始日から8日間）
- **電話勧誘販売**: 契約日から8日間

// ステートマシンによる動的な期間計算
class CoolingOffStateMachine {
  private readonly COOLING_OFF_PERIOD_DAYS = 8;
  
  async canCancelOrder(orderId: string): Promise<boolean> {
    const order = await this.orderRepository.findById(orderId);
    
    if (!order.deliveryDate) {
      return false; // 商品が未受領の場合は解約不可
    }
    
    const deliveryDate = new Date(order.deliveryDate);
    const now = new Date();
    const daysSinceDelivery = Math.floor(
      (now.getTime() - deliveryDate.getTime()) / (1000 * 60 * 60 * 24)
    );
    
    return daysSinceDelivery <= this.COOLING_OFF_PERIOD_DAYS;
  }
  
  async getRemainingDays(orderId: string): Promise<number> {
    const order = await this.orderRepository.findById(orderId);
    const deliveryDate = new Date(order.deliveryDate);
    const now = new Date();
    const daysSinceDelivery = Math.floor(
      (now.getTime() - deliveryDate.getTime()) / (1000 * 60 * 60 * 24)
    );
    
    return Math.max(0, this.COOLING_OFF_PERIOD_DAYS - daysSinceDelivery);
  }
}

    templateId: string,

      timestamp: new Date(),

    if (!checkoutData.productInfo.price.includes('（税込）')) {
      errors.push('価格に「（税込）」が含まれていません');
    }
    
    if (!checkoutData.returnPolicy.terms) {
      errors.push('返品・交換条件が表示されていません');
    }
    
    // クーリング・オフの要件をバリデーション
    if (!checkoutData.coolingOff.contactInfo.phone) {
      errors.push('クーリング・オフの連絡先（電話）が表示されていません');
    }
    
    return {
      isValid: errors.length === 0,
      errors,
    };
  }
}

export function CheckoutPage({ checkoutData }: { checkoutData: CheckoutData }) {

  if (!validationResult.isValid) {

      <SellerInfo info={checkoutData.sellerInfo} />
      <ReturnPolicy info={checkoutData.returnPolicy} />
      <CoolingOffSection info={checkoutData.coolingOff} />
    </div>
  );
}
